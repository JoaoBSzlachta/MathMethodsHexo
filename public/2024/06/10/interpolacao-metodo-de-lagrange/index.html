Método de Lagrange (Interpolação)

Acesse o Método de Lagrange e teste você mesmo.
Através do método de Lagrange é possível encontrar o polinômio interpolado de menor grau dado um determinado conjunto de pontos. Isso é possível pois cada coeficiente do polinômio é explicitamente escrito através do produto:

Assim, o polinômio interpolado de Lagrange pode ser escrito da seguinte forma:

Código em JS do método de Lagrange

function lagrangeMethod(x, y, n) {
    function L(k, n) {
        let resultado = 1;
        let termo = "";
        for (let j = 0; j < x.length; j++) {
            if (j !== k) {
                resultado *= (n - x[j]) / (x[k] - x[j]);
                if (termo) {
                    termo += " * ";
                }
                termo += `((x - ${x[j].toFixed(2)}) / (${x[k].toFixed(2)} - ${x[j].toFixed(2)}))`;
            }
        }
        return { resultado, termo };
    }

    let resultado = 0;
    let polinomio = "";
    for (let k = 0; k < x.length; k++) {
        let { resultado: Lk, termo } = L(k, n);
        if (polinomio) {
            polinomio += " + ";
        }
        polinomio += `${y[k].toFixed(6)} * ${termo}`;
        resultado += y[k] * Lk;
    }
    return { resultado, polinomio };
}
