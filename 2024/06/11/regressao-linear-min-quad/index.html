Método dos Mínimos Quadrados (Regressão Linear)

Acesse o Método dos Mínimos Quadrados e teste você mesmo.
Este método consiste em uma otimização matemática visando o melhor ajuste para um conjunto de dados e buscando a minimizar os resíduos.

Código em JS da Regressão Linear

function leastSquares(x, y){
    let n = x.length;
    let b1, b0;
    
    function somar(v){
        let result = 0;
        for (let i = 0; i < v.length; i++) {
            result += v[i];
        }
        return result;
    }

    function somarMult(u, v){
        result = 0;
        for (let i = 0; i < v.length; i++) {
            result += (u[i]*v[i]);
        }
        return result;
    }

    function somarQuad(v){
        result = 0;
        for (let i = 0; i < v.length; i++) {
            result += (v[i]**2);
        }
        return result;
    }

    b1 = ((somar(x)*somar(y)) - (n*somarMult(x, y))) / ((somar(x))**2 - (n*somarQuad(x)));

    b0 = (somar(y) - (b1*somar(x))) / n

    const ui = `${b0.toFixed(4)} + ${b1.toFixed(4)}x`;

    console.log(ui);

    return ui;
}
